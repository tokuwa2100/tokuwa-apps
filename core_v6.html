/** * CSV形式アップロード - ネットワーク制限回避版 * JSONの代わりにCSV形式でデータを送信 */ // Google Apps Script URL (既存のまま) const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbzY6CexizgJtJQaGQbIp_epGveGobN9dOMFHAk2xgKWenKFnJ6PI-l0uiR-kMcrvIgGJQ/exec"; // JSONデータをCSV形式に変換 function convertToCSV(data) { const lines = []; // ヘッダー行 lines.push("項目,値"); // 基本情報 lines.push(`日付,${data.date}`); lines.push(`お客様,${data.customer}`); lines.push(`現場名,${data.site}`); // 作業員情報 data.workers.forEach((worker, index) => { lines.push(`作業員${index + 1}_氏名,${worker.name}`); lines.push(`作業員${index + 1}_時間,${worker.hours}`); }); // コア情報 data.cores.forEach((core, index) => { lines.push(`コア${index + 1}_材質,${core.material}`); lines.push(`コア${index + 1}_直径,${core.diameter}`); lines.push(`コア${index + 1}_長さ,${core.length}`); lines.push(`コア${index + 1}_数量,${core.count}`); lines.push(`コア${index + 1}_階,${core.floor}`); }); // 備考 if (data.notes) { lines.push(`備考,${data.notes}`); } return lines.join('\n'); } // CSV形式でのアップロード関数 async function uploadCSVSelected() { const checkboxes = document.querySelectorAll('.record-checkbox:checked'); const selectedData = []; checkboxes.forEach(cb => { const index = parseInt(cb.dataset.index); selectedData.push(getRecords()[index]); }); if (selectedData.length === 0) { showToast("⚠️ データが選択されていません"); return; } try { showToast("🔄 CSV形式でアップロード中..."); for (let i = 0; i < selectedData.length; i++) { const data = selectedData[i]; const fileName = `コア日報_${data.date}_${data.site}.csv`; const csvContent = convertToCSV(data); console.log('CSV upload attempt:', fileName); console.log('CSV content preview:', csvContent.substring(0, 200)); // FormDataでCSVファイルを送信 const formData = new FormData(); formData.append('fileName', fileName); formData.append('data', csvContent); formData.append('fileType', 'csv'); formData.append('timestamp', new Date().toISOString()); const response = await fetch(SCRIPT_URL, { method: 'POST', body: formData }); console.log('Response status:', response.status); console.log('Response ok:', response.ok); if (!response.ok) { throw new Error(`CSV Upload failed: ${response.status} ${response.statusText}`); } // レスポンスの解析 const result = await response.text(); console.log('Response text:', result); let parsedResult; try { parsedResult = JSON.parse(result); } catch (e) { console.warn('Response is not JSON:', result); parsedResult = { success: true, message: 'CSV形式でアップロード完了' }; } if (!parsedResult.success) { throw new Error(parsedResult.message || 'サーバーエラー'); } } // 選択解除 checkboxes.forEach(cb => cb.checked = false); updateSelection(); showToast(`✅ ${selectedData.length}件をCSVアップロードしました`); } catch (error) { console.error('CSV Upload error:', error); showToast(`❌ CSVアップロードに失敗しました`); // 詳細エラー情報をアラート表示 alert(`CSV形式エラー詳細: エラー: ${error.message} URL: ${SCRIPT_URL} 形式: CSV (text/csv) 時刻: ${new Date().toLocaleString()}`); } } // 複数形式の段階的フォールバック（CSV優先） async function uploadWithCSVFallback() { const checkboxes = document.querySelectorAll('.record-checkbox:checked'); const selectedData = []; checkboxes.forEach(cb => { const index = parseInt(cb.dataset.index); selectedData.push(getRecords()[index]); }); if (selectedData.length === 0) { showToast("⚠️ データが選択されていません"); return; } // 複数の送信方式を順次試行（CSV優先） const methods = [ { name: 'CSV', func: uploadViaCSV }, { name: 'プレーンテキスト', func: uploadViaText }, { name: 'Base64', func: uploadViaBase64 } ]; for (const method of methods) { try { showToast(`🔄 ${method.name}形式で送信中...`); await method.func(selectedData); // 成功時 checkboxes.forEach(cb => cb.checked = false); updateSelection(); showToast(`✅ ${selectedData.length}件をアップロード（${method.name}）`); return; } catch (error) { console.log(`${method.name} failed:`, error.message); continue; } } // 全て失敗 showToast(`❌ 全ての送信形式が失敗しました`); alert(`全送信形式エラー: CSV、プレーンテキスト、Base64のすべてが失敗しました。 ネットワーク管理者にご相談ください。`); } // CSV方式（推奨） async function uploadViaCSV(selectedData) { for (const data of selectedData) { const fileName = `コア日報_${data.date}_${data.site}.csv`; const csvContent = convertToCSV(data); const formData = new FormData(); formData.append('fileName', fileName); formData.append('data', csvContent); formData.append('fileType', 'csv'); const response = await fetch(SCRIPT_URL, { method: 'POST', body: formData }); if (!response.ok) { throw new Error(`CSV upload failed: ${response.status}`); } } } // プレーンテキスト方式（代替案1） async function uploadViaText(selectedData) { for (const data of selectedData) { const fileName = `コア日報_${data.date}_${data.site}.txt`; const textContent = ` 日報データ ==================== 日付: ${data.date} お客様: ${data.customer} 現場名: ${data.site} 【作業員情報】 ${data.workers.map((w, i) => `${i+1}. ${w.name} (${w.hours}時間)`).join('\n')} 【コア作業詳細】 ${data.cores.map((c, i) => `${i+1}. φ${c.diameter} L=${c.length} ×${c.count}本 (${c.material}, ${c.floor})`).join('\n')} 備考: ${data.notes || 'なし'} `; const formData = new FormData(); formData.append('fileName', fileName); formData.append('data', textContent); formData.append('fileType', 'text'); const response = await fetch(SCRIPT_URL, { method: 'POST', body: formData }); if (!response.ok) { throw new Error(`Text upload failed: ${response.status}`); } } } // Base64方式（代替案2） async function uploadViaBase64(selectedData) { for (const data of selectedData) { const fileName = `コア日報_${data.date}_${data.site}.dat`; const jsonString = JSON.stringify(data, null, 2); const base64Data = btoa(unescape(encodeURIComponent(jsonString))); const formData = new FormData(); formData.append('fileName', fileName); formData.append('data', base64Data); formData.append('fileType', 'base64'); const response = await fetch(SCRIPT_URL, { method: 'POST', body: formData }); if (!response.ok) { throw new Error(`Base64 upload failed: ${response.status}`); } } }
